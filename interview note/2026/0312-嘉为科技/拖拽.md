# 拖拽

可以用原生 HTML5 拖拽事件来做。核心思路是：

1. 给每一行 `tr` 设置 `draggable="true"`。
2. 在 `dragstart` 记录当前被拖拽的行。
3. 在目标行的 `dragover` 里 `preventDefault()`，允许放置。
4. 在 `drop` 时比较鼠标落点在目标行的上半部分还是下半部分，决定插入到前面还是后面。

下面是一个可直接运行的完整例子：

    <!doctype html>
    <html lang="zh-CN">
    <head>
      <meta charset="UTF-8" />
      <meta name="viewport" content="width=device-width, initial-scale=1.0" />
      <title>原生表格行拖拽排序</title>
      <style>
        body {
          font-family: "Microsoft YaHei", sans-serif;
          background: #f5f7fb;
          margin: 24px;
        }

        .wrap {
          max-width: 720px;
          margin: 0 auto;
          background: #fff;
          border: 1px solid #e5e7eb;
          border-radius: 10px;
          box-shadow: 0 6px 20px rgba(0, 0, 0, 0.06);
          overflow: hidden;
        }

        .title {
          padding: 14px 16px;
          background: #0f172a;
          color: #fff;
          font-size: 16px;
          font-weight: 600;
        }

        table {
          width: 100%;
          border-collapse: collapse;
        }

        th,
        td {
          padding: 12px 14px;
          border-bottom: 1px solid #eef2f7;
          text-align: left;
          font-size: 14px;
        }

        th {
          background: #f8fafc;
          color: #334155;
        }

        tbody tr {
          cursor: grab;
          user-select: none;
          background: #fff;
          transition: background 0.15s ease;
        }

        tbody tr:hover {
          background: #f8fbff;
        }

        tbody tr.dragging {
          opacity: 0.45;
          cursor: grabbing;
        }

        tbody tr.drop-above {
          box-shadow: inset 0 2px 0 #2563eb;
        }

        tbody tr.drop-below {
          box-shadow: inset 0 -2px 0 #2563eb;
        }

        .hint {
          padding: 10px 14px 14px;
          color: #64748b;
          font-size: 13px;
        }
      </style>
    </head>
    <body>
      <div class="wrap">
        <div class="title">拖拽表格行排序示例</div>
        <table>
          <thead>
            <tr>
              <th style="width: 80px">序号</th>
              <th>任务</th>
              <th style="width: 140px">负责人</th>
            </tr>
          </thead>
          <tbody id="tableBody">
            <tr draggable="true"><td>1</td><td>需求评审</td><td>Kevin</td></tr>
            <tr draggable="true"><td>2</td><td>接口联调</td><td>Alice</td></tr>
            <tr draggable="true"><td>3</td><td>前端开发</td><td>Bob</td></tr>
            <tr draggable="true"><td>4</td><td>测试修复</td><td>Chris</td></tr>
            <tr draggable="true"><td>5</td><td>上线发布</td><td>Doris</td></tr>
          </tbody>
        </table>
        <div class="hint">按住某一行拖到目标行的上方或下方即可重新排序。</div>
      </div>

      <script>
        const tbody = document.getElementById("tableBody");
        let draggingRow = null;

        function clearDropClass() {
          [...tbody.rows].forEach((row) => {
            row.classList.remove("drop-above", "drop-below");
          });
        }

        function refreshIndex() {
          [...tbody.rows].forEach((row, idx) => {
            row.cells[0].textContent = idx + 1;
          });
        }

        [...tbody.rows].forEach((row) => {
          row.addEventListener("dragstart", () => {
            draggingRow = row;
            row.classList.add("dragging");
          });

          row.addEventListener("dragend", () => {
            row.classList.remove("dragging");
            clearDropClass();
            draggingRow = null;
            refreshIndex();
          });

          row.addEventListener("dragover", (e) => {
            if (!draggingRow || draggingRow === row) return;
            e.preventDefault();

            clearDropClass();

            const rect = row.getBoundingClientRect();
            const isAbove = e.clientY < rect.top + rect.height / 2;
            row.classList.add(isAbove ? "drop-above" : "drop-below");
          });

          row.addEventListener("drop", (e) => {
            if (!draggingRow || draggingRow === row) return;
            e.preventDefault();

            const rect = row.getBoundingClientRect();
            const isAbove = e.clientY < rect.top + rect.height / 2;

            if (isAbove) {
              tbody.insertBefore(draggingRow, row);
            } else {
              tbody.insertBefore(draggingRow, row.nextSibling);
            }

            clearDropClass();
            refreshIndex();
          });
        });
      </script>
    </body>
    </html>
