## 上传图片

### 1. 服务端配置

使用内置上传器时必填 `server` (上传图片的接口地址)，否则上传图片会报错。

如果你使用 `customUpload` 或 `uploadAdapter` ，则不需要配置 `server` 。

```ts
editorConfig.MENU_CONF['uploadImage'] = {
     server: '/api/upload',
}
```

**服务端 `response` `body` 格式要求如下**：

上传成功的返回格式：

```ts
{
    "errno": 0, // 注意：值是数字，不能是字符串
    "data": {
        "url": "xxx", // 图片 src ，必须
        "alt": "yyy", // 图片描述文字，非必须
        "href": "zzz" // 图片的链接，非必须
    }
}
```

上传失败的返回格式：

```ts
{
    "errno": 1, // 只要不等于 0 就行
    "message": "失败信息"
}
```

服务端代码，下面以 `django` 为例：

```py
from rest_framework.generics import GenericAPIView
from rest_framework.response import Response
from rest_framework import status
from django.core.files.storage import default_storage
from django.core.files.base import ContentFile
from rest_framework.parsers import MultiPartParser, FormParser
import uuid
from datetime import datetime
class ImageUploadView(GenericAPIView):
    """
    Handle wangEditor image upload interface
    """
    parser_classes = (MultiPartParser, FormParser)
    def post(self, request, *args, **kwargs):
        try:
            # Check if file is uploaded
            if "upload-image" not in request.FILES:
                return Response(
                    {"errno": 1, "message": "No uploaded image file found"}, status=status.HTTP_400_BAD_REQUEST
                )

            uploaded_file = request.FILES["upload-image"]

            # Validate file type
            allowed_extensions = [".jpg", ".jpeg", ".png", ".gif", ".bmp", ".webp"]
            file_extension = os.path.splitext(uploaded_file.name)[1].lower()
            if file_extension not in allowed_extensions:
                return Response(
                    {
                        "errno": 1,
                        "message": f'Unsupported file type. Supported formats: {", ".join(allowed_extensions)}',
                    },
                    status=status.HTTP_400_BAD_REQUEST,
                )

            # Validate file size (limit to 5MB)
            max_size = 5 * 1024 * 1024  # 5MB
            if uploaded_file.size > max_size:
                return Response(
                    {"errno": 1, "message": "File size exceeds limit (maximum 5MB)"}, status=status.HTTP_400_BAD_REQUEST
                )

            # Generate unique filename
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            unique_id = str(uuid.uuid4())[:8]
            filename = f"{timestamp}_{unique_id}{file_extension}"

            # Create save path (organize by year/month folders)
            date_path = datetime.now().strftime("%Y/%m")
            relative_path = f"uploads/images/{date_path}/{filename}"

            # Save file
            file_path = default_storage.save(relative_path, ContentFile(uploaded_file.read()))

            # Build complete URL
            if hasattr(settings, "MEDIA_URL") and settings.MEDIA_URL:
                file_url = f"{settings.MEDIA_URL}{file_path}"
            else:
                file_url = f"/media/{file_path}"

            # Add complete domain if configured
            if hasattr(settings, "DOMAIN_URL") and settings.DOMAIN_URL:
                file_url = f"{settings.DOMAIN_URL}{file_url}"

            # Return format required by wangEditor
            return Response(
                {"errno": 0, "data": {"url": file_url, "alt": uploaded_file.name, "href": file_url}},
                status=status.HTTP_200_OK,
            )
        except Exception as e:
            logger.error("Error uploading image: %s", str(e), exc_info=True)
            return Response(
                {"errno": 1, "message": f"Upload failed: {str(e)}"}, status=status.HTTP_500_INTERNAL_SERVER_ERROR
            )
```

### 2. 基本配置

```ts
editorConfig.MENU_CONF['uploadImage'] = {
    // form-data 的 fieldName ，默认值 'wangeditor-uploaded-image'
    // 后端从该字段获取上传的文件
    fieldName: 'your-custom-name',

    // 单个文件的最大体积限制，默认为 2M
    maxFileSize: 1 * 1024 * 1024, // 1M

    // 最多可上传几个文件，默认为 100
    maxNumberOfFiles: 10,

    // 选择文件时的类型限制，默认为 ['image/*'] 。如不想限制，则设置为 []
    allowedFileTypes: ['image/*'],

    // 自定义上传参数，例如传递验证的 token 等。参数会被添加到 formData 中，一起上传到服务端。
    meta: {
        token: 'xxx',
        otherKey: 'yyy'
    },

    // 将 meta 拼接到 url 参数中，默认 false
    metaWithUrl: false,

    // 自定义增加 http  header
    headers: {
        Accept: 'text/x-json',
        otherKey: 'xxx'
    },

    // 跨域是否传递 cookie ，默认为 false
    withCredentials: true,

    // 超时时间，默认为 10 秒
    timeout: 5 * 1000, // 5 秒
}
```

### 3. 回调函数

```ts
editorConfig.MENU_CONF['uploadImage'] = {
    // 上传之前触发
    onBeforeUpload(file: File) { // TS 语法
    // onBeforeUpload(file) {    // JS 语法
        // file 选中的文件，格式如 { key: file }
        return file

        // 可以 return
        // 1. return file 或者 new 一个 file ，接下来将上传
        // 2. return false ，不上传这个 file
    },

    // 上传进度的回调函数
    onProgress(progress: number) {  // TS 语法
    // onProgress(progress) {       // JS 语法
        // progress 是 0-100 的数字
        console.log('progress', progress)
    },

    // 单个文件上传成功之后
    onSuccess(file: File, res: any) {  // TS 语法
    // onSuccess(file, res) {          // JS 语法
        console.log(`${file.name} 上传成功`, res)
    },

    // 单个文件上传失败
    onFailed(file: File, res: any) {   // TS 语法
    // onFailed(file, res) {           // JS 语法
        console.log(`${file.name} 上传失败`, res)
    },

    // 上传错误，或者触发 timeout 超时
    onError(file: File, err: any, res: any) {  // TS 语法
    // onError(file, err, res) {               // JS 语法
        console.log(`${file.name} 上传出错`, err, res)
    },
}
```

### 4. 简单例子

```ts
const editorConfig = {
  placeholder: 'Please input content',
  MENU_CONF: {
    uploadImage: {
      server: import.meta.env.VITE_UPLOAD_PICTURE_URL,
      fieldName: 'upload-image',
      maxFileSize: 2 * 1024 * 1024, // 2M
      allowedFileTypes: ['image/jpeg', 'image/png', 'image/gif'],
      headers: { 'Authorization': `Bearer ${token}` },
      onFailed(file: any, res: any) {
        console.error('upload picture fail', res);
      },
      onError(file: any, err: any, res: any) {
        console.error('upload picture error', err, res);
      },
    }
  }
}
```

### 5. 自定义功能

#### 5.1. 自定义上传

使用 `customUpload()` 来自定义上传。

```ts
editorConfig.MENU_CONF['uploadImage'] = {
    // 自定义上传
    async customUpload(file: File, insertFn: InsertFnType) {
        // file 即选中的文件
        // 自己实现上传，并得到图片 url alt href
        // 最后插入图片
        insertFn(url, alt, href)
    }
}
```

例子：

```ts
const uploadPictureRequest = async (file: File) => {
  try {
    const formData = new FormData();
    // 将 file 添加到 formData 中，字段名为 'upload-image'
    formData.append('upload-image', file);
    const response = await axios.post(import.meta.env.VITE_UPLOAD_PICTURE_URL, formData, {
      headers: {
        'Content-Type': 'multipart/form-data',
      },
    });
    return response.data;
  } catch (error) {
    console.error("POST Failed:", error);
    throw error;
  }
}

editorConfig.MENU_CONF['uploadImage'] = {
    // 自定义上传
    async customUpload(file: File, insertFn: InsertFnType) {
        const { url, alt, href } = await uploadPictureRequest(file)
        insertFn(url, alt, href)
    }
}
```

> 上传图片后，如果删除图片再重新上传同一张图片会报错，原因是 `wangEditor-next` 内部的 `Uppy` 上传器会记住已经选过的文件。你把图片从编辑器内容里删掉，只是删了文档节点，没有把 `Uppy` 里的同名文件记录一起清掉，所以再次上传同名图片时就会报 `duplicate file`。
>
> 解决方法是通过 `customUpload` 来规避这个问题。
>
> 但是 `wangeditor` 没有这个问题
