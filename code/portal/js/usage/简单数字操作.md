## 十进制转 n 进制

```javascript
function convertDecimalToBaseN(number, base) {
  if (base < 2 || base > 36) {
    throw new Error("Base must be in the range 2-36");
  }
  return number.toString(base);
}
// 示例
console.log(convertDecimalToBaseN(255, 16)); // "ff"
```

## n 进制转十进制

```javascript
function convertBaseNToDecimal(numberStr, base) {
  if (base < 2 || base > 36) {
    throw new Error("Base must be in the range 2-36");
  }
  return parseInt(numberStr, base);
}
```
