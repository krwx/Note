## 拖拽整行

可以用原生 HTML5 拖拽事件来做。核心思路是：

1. 给每一行 `tr` 设置 `draggable="true"`。
2. 监听开始拖拽事件（`dragstart`），记录当前被拖拽的行，给它添加一个 `dragging` class。
3. 监听拖拽元素在当前元素上移动事件（`dragover`），清除所有行的 `drop-above` 和 `drop-below` class，通过比较鼠标位置和当前行的边界，判断是拖拽到上半部分还是下半部分，给当前行添加 `drop-above` 或 `drop-below` class 来显示放置提示。
4. 监听拖拽元素在当前元素上放置事件（`drop`），判断放置元素在上半部分还是下半部分来决定将拖拽行插入到当前行的前面还是后面，使用 `tbody.insertBefore()` 将拖拽行**移动**到目标位置。
5. 监听结束拖拽事件（`dragend`），清除 `dragging` class，重置记录的拖拽行，清除所有行的 `drop-above` 和 `drop-below` class，刷新序号显示。

> 在目标行的 `dragover` 里 `preventDefault()`，允许放置。

css 设计思路：

1. tr 的样式设置 `cursor: grab`。
2. tr 拖拽时设置 `dragging` class，样式设置 `opacity: 0.45`。
   - 此时设置 cursor 会不生效，因为拖动时浏览器会自动使用默认的拖动光标。
3. 当有 tr 拖拽到当前 tr 的上方，给当前 tr 添加 `drop-above` class，样式设置 `box-shadow: inset 0 2px 0 #2563eb`，在当前 tr 的上边缘显示一条蓝色线，提示用户放置位置。
4. 当有 tr 拖拽到当前 tr 的下方，给当前 tr 添加 `drop-below` class，样式设置 `box-shadow: inset 0 -2px 0 #2563eb`，在当前 tr 的下边缘显示一条蓝色线，提示用户放置位置。

下面是一个可直接运行的完整例子：

```html
<!doctype html>
<html lang="zh-CN">
<head>
  <meta charset="UTF-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1.0" />
  <title>原生表格行拖拽排序</title>
  <style>
    .wrap {
      max-width: 720px;
      margin: 0 auto;
      background: #fff;
      border: 1px solid #e5e7eb;
      border-radius: 10px;
      box-shadow: 0 6px 20px rgba(0, 0, 0, 0.06);
      overflow: hidden;
    }

    table {
      width: 100%;
      border-collapse: collapse;
    }

    th,
    td {
      padding: 12px 14px;
      border-bottom: 1px solid #eef2f7;
      text-align: left;
      font-size: 14px;
    }

    th {
      background: #f8fafc;
      color: #334155;
    }

    tbody tr {
      cursor: grab;
      user-select: none;
      background: #fff;
      transition: background 0.15s ease;
    }

    tbody tr:hover {
      background: #f8fbff;
    }

    tbody tr.dragging {
      opacity: 0.45;
    }

    tbody tr.drop-above {
      box-shadow: inset 0 2px 0 #2563eb;
    }

    tbody tr.drop-below {
      box-shadow: inset 0 -2px 0 #2563eb;
    }
  </style>
</head>
<body>
  <div class="wrap">
    <table>
      <thead>
        <tr>
          <th style="width: 80px">序号</th>
          <th>任务</th>
          <th style="width: 140px">负责人</th>
        </tr>
      </thead>
      <tbody id="tableBody">
        <tr draggable="true"><td>1</td><td>需求评审</td><td>Kevin</td></tr>
        <tr draggable="true"><td>2</td><td>接口联调</td><td>Alice</td></tr>
        <tr draggable="true"><td>3</td><td>前端开发</td><td>Bob</td></tr>
        <tr draggable="true"><td>4</td><td>测试修复</td><td>Chris</td></tr>
        <tr draggable="true"><td>5</td><td>上线发布</td><td>Doris</td></tr>
      </tbody>
    </table>
  </div>

  <script>
    const tbody = document.getElementById("tableBody");
    let draggingRow = null;

    function clearDropClass() {
      [...tbody.rows].forEach((row) => {
        row.classList.remove("drop-above", "drop-below");
      });
    }

    function refreshIndex() {
      [...tbody.rows].forEach((row, idx) => {
        row.cells[0].textContent = idx + 1;
      });
    }

    [...tbody.rows].forEach((row) => {
      row.addEventListener("dragstart", () => {
        draggingRow = row;
        row.classList.add("dragging");
      });

      row.addEventListener("dragend", () => {
        row.classList.remove("dragging");
        draggingRow = null;
        clearDropClass();
        refreshIndex();
      });

      row.addEventListener("dragover", (e) => {
        if (!draggingRow || draggingRow === row) return;
        e.preventDefault();

        clearDropClass();

        const rect = row.getBoundingClientRect();
        const isAbove = e.clientY < rect.top + rect.height / 2;
        row.classList.add(isAbove ? "drop-above" : "drop-below");
      });

      row.addEventListener("drop", (e) => {
        e.preventDefault();
        if (!draggingRow || draggingRow === row) return;
        e.preventDefault();

        const rect = row.getBoundingClientRect();
        const isAbove = e.clientY < rect.top + rect.height / 2;

        if (isAbove) {
          tbody.insertBefore(draggingRow, row);
        } else {
          tbody.insertBefore(draggingRow, row.nextSibling);
        }
      });
    });
  </script>
</body>
</html>
```

## 点击操作列进行拖拽

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>原生表格 · 拖拽行排序（手柄拖动）</title>
    <style>
        body {
            display: flex;
            justify-content: center;
            align-items: center;
            min-height: 100vh;
            margin: 0;
            padding: 16px;
        }
        .container {
            width: 650px;
            max-width: 100%;
            background: white;
            border-radius: 20px;
            box-shadow: 0 20px 40px -10px rgba(0,20,40,0.2);
            padding: 24px 20px 28px;
        }
        table {
            width: 100%;
            border-collapse: collapse;

        /* 拖拽手柄列样式 */
        .drag-handle {
            width: 48px;

            cursor: grab;
            font-size: 1.5rem;

            user-select: none;
            transition: background 0.2s, color 0.2s;
        }
        .drag-handle:hover {
            background: #e6edf8;
            color: #2f4b7c;
        }
        tr:active .drag-handle {
            cursor: grabbing;
        }
        /* 正在拖拽的行 (半透明效果) */
        tr.dragging {
            opacity: 0.6;
            box-shadow: 0 4px 8px rgba(0,0,0,0.05);
        }
        /* 拖拽过程中的放置目标高亮 */
        tr.drag-over {
            background-color: #eef4ff;
            box-shadow: inset 0 2px 0 #7495cf, inset 0 -2px 0 #7495cf;
        }

<body>
<div class="container">
    <h2>
        可拖拽排序的表格
        <span>仅抓手区域可拖动</span>

                <td class="drag-handle">☰</td>

                <td class="value-col">19.7 KB</td>
            </tr>
            <tr draggable="true" data-id="5">
                <td class="drag-handle">☰</td>

                <td class="value-col">43.2 KB</td>
            </tr>
        </tbody>

<script>
    (function() {
        // 获取表格主体
        const tbody = document.getElementById('table-body');
        // 记录是否由手柄开始本次拖拽，避免误拖整行
        let handleDragRowId = null;

        tbody.addEventListener('mousedown', (e) => {
            // 只有点击手柄才能拿到 handleCell，然后才记录拖拽的行 ID
            const handleCell = e.target.closest('.drag-handle');
            const row = e.target.closest('tr');
            handleDragRowId = handleCell && row ? row.getAttribute('data-id') : null;
        });

        document.addEventListener('mouseup', () => {
            handleDragRowId = null;
        });

        // --- 清理高亮与拖拽状态的辅助函数 ---
        function clearAllDragStyles() {
            const rows = tbody.querySelectorAll('tr');
            rows.forEach(row => {
                row.classList.remove('dragging', 'drag-over');
            });
        }

        // --- dragstart：开始拖拽，只有点击手柄列才允许拖拽 ---
        tbody.addEventListener('dragstart', (e) => {
            console.log(e)
            // 对于 draggable="true" 的 tr，dragstart 的 target 通常是 tr，而不是 td
            const row = e.target.closest('tr');
            if (!row) {
                e.preventDefault();
                return;
            }

            // 仅允许“从手柄按下后”开始的拖拽
            const rowId = row.getAttribute('data-id');
            if (!rowId || handleDragRowId !== rowId) {
                e.preventDefault();
                return;
            }

            // 设置拖拽数据与效果
            e.dataTransfer.setData('text/plain', rowId);
            e.dataTransfer.effectAllowed = 'move';

            // 添加拖拽样式（便于视觉反馈）
            row.classList.add('dragging');
        });

        // --- dragover：必须阻止默认，并显示可放置的高亮 ---
        tbody.addEventListener('dragover', (e) => {
            e.preventDefault();               // 必须！
            e.dataTransfer.dropEffect = 'move';

            // 获取鼠标下方的行
            const targetRow = e.target.closest('tr');
            if (!targetRow) return;

            // 先移除所有行的高亮，再给当前行加上高亮 (实现干净切换)
            tbody.querySelectorAll('tr').forEach(r => r.classList.remove('drag-over'));
            targetRow.classList.add('drag-over');
        });

        // --- dragleave：移除当前行的高亮（如果鼠标离开该行）---
        tbody.addEventListener('dragleave', (e) => {
            // 只有当鼠标真的离开整个表格行区域时才移除高亮
            // 使用 relatedTarget 判断是否离开了行
            const targetRow = e.target.closest('tr');
            if (!targetRow) return;

            const related = e.relatedTarget;
            if (!related || !targetRow.contains(related)) {
                targetRow.classList.remove('drag-over');
            }
        });

        // --- drop：放置处理，重新排序 ---
        tbody.addEventListener('drop', (e) => {
            e.preventDefault();  // 必须阻止默认行为（如打开链接）

            // 获取目标行（鼠标放开时所在的行）
            const targetRow = e.target.closest('tr');
            if (!targetRow) return;   // 没有放在行上，忽略

            // 获取被拖拽行的 ID（从 dataTransfer 中）
            const draggedId = e.dataTransfer.getData('text/plain');
            if (!draggedId) return;

            // 通过 ID 查找到被拖拽的行（注意：此时行可能还在原位置）
            const draggedRow = document.querySelector(`tr[data-id="${draggedId}"]`);
            if (!draggedRow) return;

            // 禁止将行拖放到自己身上（无意义）
            if (draggedRow === targetRow) {
                // 清除样式并返回
                clearAllDragStyles();
                return;
            }

            // 获取目标行矩形及鼠标垂直位置，决定插入到目标行的上方还是下方
            const targetRect = targetRow.getBoundingClientRect();
            const mouseY = e.clientY;
            const insertBeforeTarget = mouseY < targetRect.top + targetRect.height / 2;

            // 获取父节点（即 tbody）
            const parent = targetRow.parentNode;
            if (!parent) return;

            // 执行移动：根据判断插入到目标行的前面或后面
            if (insertBeforeTarget) {
                // 插入到目标行之前
                parent.insertBefore(draggedRow, targetRow);
            } else {
                // 插入到目标行之后 (nextSibling 可能为 null，insertBefore 会正确处理)
                parent.insertBefore(draggedRow, targetRow.nextSibling);
            }

            // 拖拽放置完成，清理所有高亮及拖拽样式
            clearAllDragStyles();
        });

        // --- dragend：拖拽完全结束（无论成功放置还是取消），清理所有样式 ---
        tbody.addEventListener('dragend', (e) => {
            clearAllDragStyles();
            handleDragRowId = null;
        });
    })();
</script>
</body>
</html>
```
